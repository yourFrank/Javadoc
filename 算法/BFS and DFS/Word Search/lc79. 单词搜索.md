package com.fisec.vpn.util;


import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;


class Solution {
    public boolean exist(char[][] board, String word) {
        //每个单元格开始进行回溯搜索
        int m = board.length;
        int n = board[0].length;
        boolean visited[][] = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                boolean flag = check(i, j, board, visited, word, 0);
                if (flag) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean check(int i, int j, char[][] board, boolean[][] visited, String word, int index) {
        if (board[i][j] != word.charAt(index)) {
            return false;
        }
        //上面if没走，此时说明相等了，如果长度也相等可以返回true
        if (index == word.length() - 1) {
            return true;
        }
        visited[i][j] = true;
        int direction[][] = {{0, 1}, {-1, 0}, {1, 0}, {0, -1}};
        //相邻 的四个位置都判断一下
        boolean result = false;
        for (int[] dir : direction) {
            int newi = i + dir[0];
            int newj = j + dir[1];
            //保证新的没有越界
            if (newi >= 0 && newi < board.length && newj >= 0 && newj < board[0].length) {
                //如果没有访问过再访问
                if (!visited[newi][newj]) {
                    result = check(newi, newj, board, visited, word, index + 1);
                    if (result) {
                        break;
                    }
                }
            }
        }
        visited[i][j] = false;
        return result;
    }
}
