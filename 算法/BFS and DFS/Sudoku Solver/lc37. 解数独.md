package com.fisec.vpn.util;


//解数独
class Solution {
    public void solveSudoku(char[][] board) {
        dfs(board, 0, 0);
    }

    private boolean dfs(char[][] board, int i, int j) {
        //设置成越界的值。范围为0-8，如果到了9说明已经越界了
        int m = 9, n = 9;
        if (j == n) {
            return dfs(board, i + 1, 0);
        }
        //如果已经越界说明完成了，直接返回true。 这里只要求一个结果就行
        if (i == m) {
            return true;
        }
        if (board[i][j] != '.') {
            return dfs(board, i, j + 1);
        }
        for (char ch = '1'; ch <= '9'; ch++) {
            //如果填写当前这个数字到i,j位置不行的话就继续
            if (!isValid(board, i, j, ch)) {
                continue;
            }

            board[i][j] = ch;
            //如果找到了一个可行解立马结束
            if (dfs(board, i, j + 1)) {
                return true;
            }
            board[i][j] = '.';
        }
        return false;
    }


    // 判断 board[i][j] 是否可以填入 n
    boolean isValid(char[][] board, int i, int j, char n) {
        for (int k = 0; k < 9; k++) {
            // 判断行是否存在重复
            if (board[i][k] == n) return false;
            // 判断列是否存在重复
            if (board[k][j] == n) return false;
            // 判断 3 x 3 方框是否存在重复
            if (board[(i / 3) * 3 + k / 3][(j / 3) * 3 + k % 3] == n)
                return false;
        }
        return true;
    }
}
