package com.fisec.vpn.util;

import java.util.*;


class Solution {
    public String minWindow(String s, String t) {
        //window表示窗口中在子串t已经出现字符的频次
        //need表示子串中每个字符需要出现的频次
        Map<Character, Integer> need = new HashMap<>();
        Map<Character, Integer> window = new HashMap<>();
        char[] chars = t.toCharArray();
        //首先将需要字符都放入need中
        for (char c : chars) {
            need.put(c, need.getOrDefault(c, 0) + 1);
        }

        //valid用来标识满足的need个数
        //[left,right）窗口不断扩大，一直到need.size=valid
        int left = 0, right = 0, valid = 0, start = 0, len = Integer.MAX_VALUE;
        while (right < s.length()) {
            char c = s.charAt(right);//当前要加入到窗口的字符
            right++;
            //进行窗口内数据的更新
            if (need.containsKey(c)) {
                window.put(c, window.getOrDefault(c, 0) + 1);
                //如果该字符在窗口中出现的次数已经满足了需要出现的次数，此时valid++
                if (window.get(c).equals(need.get(c))) //这里注意要用equals判断
                    valid++;
            }
            //判断左侧窗口是否需要收缩
            while (valid == need.size()) {
                if (right - left < len) {
                    start = left;
                    len = right - left;
                }
                char b = s.charAt(left);
                left++;
                //进行窗口内数据的更新
                if (need.containsKey(b)) {
                    //如果在need中需要出现的频次和窗口中目前的值一样，该值被移除了。就要减少valid
                    if (window.get(b).equals(need.get(b)))//如果本身window中的b和need中的b的次数是一样的，这时移除了就要valid--
                        valid--;
                    window.put(b, window.get(b) - 1);
                }

            }
        }
        return len == Integer.MAX_VALUE ? "" : s.substring(start, start + len);

    }
}
