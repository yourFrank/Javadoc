package com.fisec.vpn.util;

class ListNode {
    ListNode next;
    int val;

    ListNode() {
    }

    ListNode(int val, ListNode next) {
        this.val = val;
        this.next = next;
    }
    ListNode(int val) { this.val = val; }
}



//k个反转链表递归实现，
class Solution {

    public ListNode reverseKGroup(ListNode head, int k) {

        ListNode cur=head;
        for (int i = 0; i < k; i++) {
            //递归终止
            if (cur==null){
                return head;
            }
            cur=cur.next;
        }

        //反转当前的链表，
        ListNode newHead = reverseN(head, k);
        //反转后面的
        head.next=reverseKGroup(cur, k);

        return newHead;

    }
//    反转head开始的k个元素
    private ListNode reverseN(ListNode head, int k) {
        if (k==1){
            return head;
        }

        ListNode listNode = reverseN(head.next, k - 1);
        head.next.next=head;
        return listNode;
    }
}
