package com.fisec.vpn.util;


import java.util.ArrayList;
import java.util.List;




//1614括号最大深度，题目已知给定了一个有效的括号
class Solution {
    public int maxDepth(String s) {
        int max = 0;
        int cnt = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i)=='('){
                cnt++;
                max=Math.max(max,cnt);
            }else if (s.charAt(i)==')'){
                cnt--;
            }
        }
        return max;
    }
}
