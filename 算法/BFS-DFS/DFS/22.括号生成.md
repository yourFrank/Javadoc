package com.fisec.vpn.util;


import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;




class Solution {

    public List<String> generateParenthesis(int n) {
        List<String>  res = new ArrayList<>();
        backtrack(res,new StringBuilder(), n, n);
        return res;
    }

    /**
     *
     * @param res
     * @param sb
     * @param l 记录左括号剩余的数量
     * @param r 记录右括号剩余的数量
     *    利用l和r的数量来判断括号是否是有效的
     */
    private void backtrack(List<String> res,StringBuilder sb, int l, int r) {
        //必要的剪枝操作
        //如果已经使用的右括号数量小于左括号的数量，此时不合法了。
        if (r < l) return;

        //此时都添加完了，并且是有效的括号
        if (l == 0 && r == 0) {
            res.add(sb.toString());
            return;
        }

        if (l>0){
            sb.append('(');
            backtrack(res,sb, l - 1, r);
            sb.deleteCharAt(sb.length() - 1);
        }
        if (r>0){
            sb.append(')');
            backtrack(res,sb, l, r - 1);
            sb.deleteCharAt(sb.length() - 1);
        }

    }
}
