package com.fisec.vpn.util;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;




class Solution {
    public int reversePairs(int[] nums) {
        return countPairs(nums, 0, nums.length - 1);
    }

    public int countPairs(int[] nums, int l, int r) {
        if (l >= r) {
            return 0;
        }
        int mid = l + (r - l) / 2;
        int leftCount = countPairs(nums, l, mid);
        int rightCount = countPairs(nums, mid + 1, r);
        int mergeCount = merge(nums, l, r);
        return leftCount + rightCount + mergeCount;

    }

    public int merge(int[] nums, int l, int r) {
        int mid = (l + r) / 2;
        int i = l, j = mid + 1, k = 0, count = 0;
        int temp[] = new int[r - l + 1];
        while (i <= mid && j <= r) {
            if (nums[i] <= nums[j]) {
                temp[k++] = nums[i++];
            } else {
                count += (mid - i + 1); //只需要在归并排序中加这一行代码。
                temp[k++] = nums[j++];
            }
        }
        while (i <= mid) {
            temp[k++] = nums[i++];
        }
        while (j <= r) {
            temp[k++] = nums[j++];
        }
        for (i = l, j = 0; i <= r; i++, j++) {
            nums[i] = temp[j];
        }
        return count;

    }

}
