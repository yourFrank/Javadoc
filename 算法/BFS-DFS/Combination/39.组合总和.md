package com.fisec.vpn.util;

import java.util.*;



class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> res = new ArrayList<>();
        dfs(res, new LinkedList<>(), candidates, target, 0, 0);
        return res;
    }

    public void dfs(List<List<Integer>> res, LinkedList<Integer> path, int[] candidates, int target, int sum, int begin) {
        // 找到了数字和为 target 的组合
        if (sum == target) {
            res.add(new ArrayList<>(path));
            return;
        }else if (sum>target){
            return;
        }
        //这里很关键，注意为了避免重复，记录上次的i，从i开始。
        for (int i = begin; i < candidates.length; i++) {
            path.addLast(candidates[i]);
            dfs(res, path, candidates, target, sum + candidates[i], i);
            path.removeLast(); // 回溯，移除路径 path 最后一个元素
        }
    }
}


class Solution2 {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> res = new ArrayList<>();
        dfs(res, new LinkedList<>(), candidates, target, 0, 0);
        return res;
    }

    public void dfs(List<List<Integer>> res, LinkedList<Integer> path, int[] candidates, int target, int sum, int begin) {
        // 找到了数字和为 target 的组合
        if (sum == target) {
            res.add(new ArrayList<>(path));
            return;
        }else if (sum>target){
            return;
        }

        for (int i = 0; i < candidates.length; i++) {
            //这里数组排序后可以剪枝优化，如果加入当前的数已经大于target,后面的大数都会大，此时break
            if (sum+candidates[i]>target){
                break;
            }
            path.addLast(candidates[i]);
            dfs(res, path, candidates, target, sum + candidates[i], i);
            path.removeLast(); // 回溯，移除路径 path 最后一个元素
        }
    }
}
