package test.com.fisec.vpn.util;


import java.awt.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

//官方写法，先序裁剪
class Solution {

    //利用二叉搜索树的性质裁剪并且返回根节点，递归
    public TreeNode trimBST(TreeNode root, int low, int high) {
        if (root == null) {
            return null;
        }

        if (root.val > high) {
            return trimBST(root.left, low, high);
        }
        if (root.val < low) {
            return trimBST(root.right, low, high);
        }
        //否则两边都要裁剪
        root.left = trimBST(root.left, low, high);
        root.right = trimBST(root.right, low, high);
        return root;
    }
}

//这是我最开始想的写法，从叶子节点开始裁剪。并没用用到二叉搜索树的性质。。。
class Solution2 {
    public TreeNode trimBST(TreeNode root, int low, int high) {
        if (root == null) {
            return null;
        }
        TreeNode leftNode = trimBST(root.left, low, high);
        TreeNode rightNode = trimBST(root.right, low, high);
        if (root.val >= low && root.val <= high) {
            root.left = leftNode;
            root.right = rightNode;
            return root;
        } else if (leftNode != null) {
            TreeNode node = leftNode;
            while (node.right != null) {
                node = node.right;
            }
            node.right = rightNode;
            return leftNode;
        } else {
            return rightNode;
        }
    }
}
