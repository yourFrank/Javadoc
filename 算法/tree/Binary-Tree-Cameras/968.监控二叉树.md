package com.fisec.vpn.util;

import java.util.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}


class Solution {
    int count = 0;

    public int minCameraCover(TreeNode root) {
        // 如果是空树，不需要
        if (root == null)
            return 0;
        // 只有根节点一个，只需要一个
        if (root.left == null && root.right == null)
            return 1;
        // 根据返回值，检查根节点是否被覆盖
        int val = helper(root);
        //最后如果返回0说明最后一个根节点没有覆盖，需要再加1
        if (val == 0)
            count += 1;
        return count;
    }

    /**
     * return
     * 0 : 表示没有覆盖到
     * 1 : 覆盖到了，但是该节点没有摄像头
     * 2 : 覆盖到了，且该节点有摄像头
     */
    public int helper(TreeNode root) {
        // 空节点，当做覆盖到了
        if (root == null) {
            return 1;
        }
        // 叶节点，贪心的策略：使用其父节点放置摄像头，能覆盖更多的节点
        if (root.left == null && root.right == null) {
            return 0;
        }
        // 后续遍历
        int left = helper(root.left);
        int right = helper(root.right);
        // 子节点有没有覆盖到的，必须在此处放置摄像头
        if (left == 0 || right == 0) {
            count++;
            return 2;
        }
        // 子节点里有任意一个放置了摄像头，此处不需要重复放置，因为已经被覆盖
        if (left == 2 || right == 2) {
            return 1;
        }
        // 子节点两个都被覆盖到了，但没有放置摄像头，因此当前点没有覆盖
        return 0;
    }
}
