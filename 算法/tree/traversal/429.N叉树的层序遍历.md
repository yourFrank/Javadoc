package test.com.fisec.vpn.util;

import java.util.ArrayList;
import java.util.Deque;
import java.util.LinkedList;
import java.util.List;

// Definition for a Node.
class Node {
    public int val;
    public List<Node> children;

    public Node() {}

    public Node(int _val) {
        val = _val;
    }

    public Node(int _val, List<Node> _children) {
        val = _val;
        children = _children;
    }
};

/**
 * 最好的解法：递归写法，时间复杂度On ，空间复杂度O(logn),最坏空间复杂度为On（当n叉树平衡的时候层数为logn，此时递归栈的空间为logn。最坏情况退化为链表n层）
 *
 */
class Solution {

    //result中的索引对应了每层的层数，因此我们可以在递归的时候记录层数，将当前层数的node获取到result中的索引list添加进去
    private List<List<Integer>> result = new ArrayList<>();

    public List<List<Integer>> levelOrder(Node root) {
        if (root != null) {
            traverseNode(root, 0);
        }
        return result;
    }
//[1,null,3,2,4,null,5,6]
    private void traverseNode(Node node, int level) {
        //这里注意当size等于level的时候，因为level是从0开始的。比如size=1对应level是0，当level=1时此时就要在集合中new新的层数
        if (result.size()<=level){
            result.add(new ArrayList<>());
        }
        result.get(level).add(node.val);
        for (Node child : node.children) {
            traverseNode(child,level+1);
        }

    }


}

/**
 * 层序遍历 时间复杂度O(n)，空间复杂度O(n)，要保存所有的节点
 */
class Solution2 {
    public List<List<Integer>> levelOrder(Node root) {
        Deque<Node> deque=new LinkedList<>();
        List<List<Integer>> res=new ArrayList<>();
        if (root!=null){
            deque.add(root);
            while (!deque.isEmpty()){
                List<Integer> list=new ArrayList<>();
                int size=deque.size();
                for (int i=0;i<size;i++){
                    Node poll = deque.poll();
                    list.add(poll.val);
                    // for (Node child : poll.children) {
                    //     deque.add(child);
                    // }
                    deque.addAll(poll.children); //这里可以直接调用addAll
                }
                res.add(list);
            }

        }
        return res;
    }
}
