package com.fisec.vpn.util;


import java.time.Period;
import java.util.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}


class Solution {
    List<String> res = new ArrayList<>();

    public List<String> binaryTreePaths(TreeNode root) {
        dfs(root, "");
        return res;
    }

    private void dfs(TreeNode root, String sb) {
        if (root==null){
            return;
        }
            StringBuilder sbr = new StringBuilder(sb);
            sbr.append(root.val);
            if (root.left == null && root.right == null) {
                res.add(sbr.toString());
            }else{
                sbr.append("->");
                dfs(root.left, sbr.toString());
                dfs(root.right, sbr.toString());
            }

    }
}
