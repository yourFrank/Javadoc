package com.fisec.vpn.util;

import com.alibaba.druid.sql.visitor.functions.Left;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

//dfs
class Solution {
    public TreeNode searchBST(TreeNode root, int val) {
        if (root==null){
            return null;
        }
        if (val > root.val) {
            TreeNode left = searchBST(root.right, val);
            return left;
        } else if (val < root.val) {
            TreeNode right = searchBST(root.left, val);
            return right;
        } else {
            return root;
        }

    }
}
//迭代
class Solution2 {
    public TreeNode searchBST(TreeNode root, int val) {
        while (root != null) {
            if (val == root.val) {
                return root;
            }
            root = val < root.val ? root.left : root.right;
        }
        return null;
    }
}
