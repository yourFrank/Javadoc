package com.fisec.vpn.util;

import org.apache.commons.collections.map.HashedMap;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

//删除二叉搜索树中的节点
//把后面的值或者前面的值替代当前删除节点的值
class Solution {
    /*
    One step right and then always left
    */
    public int successor(TreeNode root) {
        root = root.right;
        while (root.left != null) root = root.left;
        return root.val;
    }

    /*
    One step left and then always right
    */
    public int predecessor(TreeNode root) {
        root = root.left;
        while (root.right != null) root = root.right;
        return root.val;
    }
    //删除当前子树中值为key的节点，并且返回删除后新的根节点
    public TreeNode deleteNode(TreeNode root, int key) {
        if (root == null) return null;

        // delete from the right subtree
        if (key > root.val) root.right = deleteNode(root.right, key);
            // delete from the left subtree
        else if (key < root.val) root.left = deleteNode(root.left, key);
            // delete the current node
        else {
            // the node is a leaf
            if (root.left == null && root.right == null) root = null;
                // the node is not a leaf and has a right child
            else if (root.right != null) {
                root.val = successor(root);//将root.val换为后继节点值
                root.right = deleteNode(root.right, root.val);//这里注意删除完成后返回新的根节点要连起来
            }
            // the node is not a leaf, has no right child, and has a left child
            else {
                root.val = predecessor(root);//将root.val换为前继节点值
                root.left = deleteNode(root.left, root.val);//这里注意删除完成后返回新的根节点要连起来
            }
        }
        return root;
    }
}
