package com.fisec.vpn.util;

import com.alibaba.druid.sql.visitor.functions.Left;

import java.util.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}


/**
 中序遍历二叉搜索树得到的就是有序数组。
 */
class Solution {
    private int cnt;
    private TreeNode kthNode;
    public int kthSmallest(TreeNode root, int k) {
        dfs(root, k);
        return kthNode.val;
    }

    private void dfs(TreeNode root, int k) {
        if (root == null) return;
        dfs(root.left, k);
        if (kthNode != null) return;
        if (++cnt == k) {
            kthNode = root;
            return;
        }
        dfs(root.right, k);
    }

}
//手动模拟栈
class Solution2 {
    public int kthSmallest(TreeNode root, int k) {
        Deque<TreeNode> stack = new ArrayDeque<TreeNode>();
        while (root != null || !stack.isEmpty()) {
            while (root != null) {
                stack.push(root);
                root = root.left;
            }
            root = stack.pop();
            --k;
            if (k == 0) {
                break;
            }
            root = root.right;
        }
        return root.val;
    }
}
