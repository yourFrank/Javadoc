package com.fisec.vpn.util;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(int val) {
        this.val = val;
    }

    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

//二叉搜索树的最小值之差
class Solution {
    private int pre = -1;
    private int min = Integer.MAX_VALUE;

    public int getMinimumDifference(TreeNode root) {
        if (root == null) {
            return 0;
        }
        getMinimumDifference(root.left);
        if (pre >= 0) {
            min = Math.min(Math.abs(root.val - pre), min);
        }
        pre = root.val;
        getMinimumDifference(root.right);
        return min;
    }

}
